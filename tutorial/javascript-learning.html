üöÄ JavaScript Overview

What is JavaScript?

Definition: JavaScript is a high-level, interpreted programming language that enables interactive web pages and dynamic content. It's the programming language of the web.

Key Features:

  • Dynamic: Variables can change types at runtime
  • Interpreted: No compilation step required
  • Event-driven: Responds to user interactions
  • Prototype-based: Object-oriented through prototypes
  • First-class functions: Functions are values
  • Asynchronous: Non-blocking operations support

JavaScript Environments:

// Browser Environment
console.log('Running in browser');
document.getElementById('myElement');
window.location.href = 'https://example.com';

// Node.js Environment
const fs = require('fs');
const path = require('path');
process.env.NODE_ENV;

// Modern Module System
import { myFunction } from './module.js';
export default myClass;

Including JavaScript in HTML:

<script src="script.js"></script>

<!-- Internal JavaScript -->
<script>
    console.log('Hello, World!');
</script>

<!-- Inline JavaScript (Avoid) -->
<button onclick="alert('Clicked!')">Click me</button>

<!-- Modern module loading -->
<script type="module" src="app.js"></script>

Advanced Function Concepts

// Higher-order functions
function createMultiplier(factor) {
    return function(number) {
        return number * factor;
    };
}

const double = createMultiplier(2);
const triple = createMultiplier(3);

console.log(double(5));  // 10
console.log(triple(5));  // 15

// Callback functions
function processArray(arr, callback) {
    const result = [];
    for (let i = 0; i < arr.length; i++) {
        result.push(callback(arr[i], i));
    }
    return result;
}

const numbers = [1, 2, 3, 4, 5];
const doubled = processArray(numbers, (num, index) => num * 2);
console.log(doubled); // [2, 4, 6, 8, 10]

// Closures
function createCounter() {
    let count = 0;
    
    return {
        increment: () => ++count,
        decrement: () => --count,
        getValue: () => count
    };
}

const counter = createCounter();
console.log(counter.increment()); // 1
console.log(counter.increment()); // 2
console.log(counter.getValue());  // 2

// IIFE (Immediately Invoked Function Expression)
(function() {
    console.log('IIFE executed immediately');
})();

// Arrow function IIFE
(() => {
    console.log('Arrow IIFE executed');
})();

// Function methods: call, apply, bind
const person = {
    name: 'John',
    greet: function(greeting, punctuation) {
        return `${greeting}, I'm ${this.name}${punctuation}`;
    }
};

const anotherPerson = { name: 'Jane' };

// call - invoke immediately with specific 'this'
console.log(person.greet.call(anotherPerson, 'Hello', '!')); // "Hello, I'm Jane!"

// apply - like call but with array of arguments
console.log(person.greet.apply(anotherPerson, ['Hi', '.'])); // "Hi, I'm Jane."

// bind - create new function with bound 'this'
const boundGreet = person.greet.bind(anotherPerson);
console.log(boundGreet('Hey', '?')); // "Hey, I'm Jane?"

Async/Await

// Async function declaration
async function fetchUserData(userId) {
    try {
        const response = await fetch(`https://jsonplaceholder.typicode.com/users/${userId}`);
        
        if (!response.ok) {
            throw new Error(`HTTP error! status: ${response.status}`);
        }
        
        const userData = await response.json();
        return userData;
    } catch (error) {
        console.error('Error fetching user data:', error);
        throw error; // Re-throw to allow caller to handle
    }
}

// Using async function
async function displayUser() {
    try {
        const user = await fetchUserData(1);
        console.log('User:', user.name);
        console.log('Email:', user.email);
    } catch (error) {
        console.log('Failed to display user');
    }
}

displayUser();

// Async arrow function
const fetchPosts = async () => {
    const response = await fetch('https://jsonplaceholder.typicode.com/posts');
    const posts = await response.json();
    return posts.slice(0, 5); // Return first 5 posts
};

// Parallel async operations
async function fetchMultipleData() {
    try {
        // Sequential (slower)
        const user = await fetchUserData(1);
        const posts = await fetchPosts();
        
        // Parallel (faster)
        const [userParallel, postsParallel] = await Promise.all([
            fetchUserData(1),
            fetchPosts()
        ]);
        
        console.log('Sequential user:', user.name);
        console.log('Parallel user:', userParallel.name);
        console.log('Posts count:', postsParallel.length);
    } catch (error) {
        console.error('Error in parallel fetch:', error);
    }
}

// Error handling with async/await
async function robustFetch(url) {
    const maxRetries = 3;
    let retries = 0;
    
    while (retries < maxRetries) {
        try {
            const response = await fetch(url);
            if (response.ok) {
                return await response.json();
            }
            throw new Error(`HTTP ${response.status}`);
        } catch (error) {
            retries++;
            if (retries === maxRetries) {
                throw new Error(`Failed after ${maxRetries} retries: ${error.message}`);
            }
            
            // Wait before retry
            await new Promise(resolve => setTimeout(resolve, 1000 * retries));
        }
    }
}

// Async iteration
async function processItems(items) {
    for (const item of items) {
        await processItem(item); // Process items sequentially
    }
}

async function processItemsParallel(items) {
    const promises = items.map(item => processItem(item));
    await Promise.all(promises); // Process items in parallel
}

// Async generators
async function* fetchPages(baseUrl) {
    let page = 1;
    let hasMore = true;
    
    while (hasMore) {
        const response = await fetch(`${baseUrl}?page=${page}`);
        const data = await response.json();
        
        if (data.length === 0) {
            hasMore = false;
        } else {
            yield data;
            page++;
        }
    }
}

// Using async generator
async function processAllPages() {
    for await (const pageData of fetchPages('https://api.example.com/items')) {
        console.log('Processing page with', pageData.length, 'items');
    }
}

Q1: What's the difference between let, const, and var?

Answer:

  • var: Function-scoped, hoisted, can be redeclared
  • let: Block-scoped, hoisted but not initialized, cannot be redeclared
  • const: Block-scoped, must be initialized, immutable binding

Q2: Explain closures in JavaScript

Answer: A closure is when an inner function has access to variables from its outer function's scope even after the outer function has returned. Closures are created every time a function is created.

Q3: What's the difference between == and ===?

Answer:

  • ==: Loose equality, performs type coercion
  • ===: Strict equality, no type coercion, checks type and value
  • Best practice: Always use === unless you specifically need coercion

Q4: What is event delegation?

Answer: Event delegation is attaching an event listener to a parent element to handle events for its children, including dynamically added ones. Uses event bubbling to catch events from child elements.

Q5: Explain the difference between Promise and async/await

Answer: Promises use .then() chains for handling asynchronous operations. Async/await is syntactic sugar over Promises, making asynchronous code look synchronous and easier to read and debug.
