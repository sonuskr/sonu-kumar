Basic Number Operations

1. Reverse a Number

public class ReverseNumber {
    public static int reverseNumber(int num) {
        int reversed = 0;
        
        while (num != 0) {
            reversed = reversed * 10 + num % 10;
            num /= 10;
        }
        
        return reversed;
    }
    
    public static void main(String[] args) {
        int num = 12345;
        System.out.println("Original: " + num);
        System.out.println("Reversed: " + reverseNumber(num));
    }
}

2. Sum of Digits

public class SumOfDigits {
    public static int sumOfDigits(int num) {
        int sum = 0;
        
        while (num > 0) {
            sum += num % 10;
            num /= 10;
        }
        
        return sum;
    }
    
    // Recursive approach
    public static int sumOfDigitsRecursive(int num) {
        if (num == 0) return 0;
        return (num % 10) + sumOfDigitsRecursive(num / 10);
    }
    
    public static void main(String[] args) {
        int num = 12345;
        System.out.println("Sum of digits: " + sumOfDigits(num));
        System.out.println("Sum (recursive): " + sumOfDigitsRecursive(num));
    }
}

3. Count Digits

public class CountDigits {
    public static int countDigits(int num) {
        if (num == 0) return 1;
        
        int count = 0;
        while (num > 0) {
            count++;
            num /= 10;
        }
        
        return count;
    }
    
    // Using logarithm
    public static int countDigitsLog(int num) {
        if (num == 0) return 1;
        return (int) Math.floor(Math.log10(num)) + 1;
    }
    
    public static void main(String[] args) {
        int num = 12345;
        System.out.println("Number of digits: " + countDigits(num));
        System.out.println("Using log: " + countDigitsLog(num));
    }
}

4. GCD and LCM

public class GcdLcm {
    public static int gcd(int a, int b) {
        if (b == 0) return a;
        return gcd(b, a % b);
    }
    
    public static int lcm(int a, int b) {
        return (a * b) / gcd(a, b);
    }
    
    // Iterative GCD
    public static int gcdIterative(int a, int b) {
        while (b != 0) {
            int temp = b;
            b = a % b;
            a = temp;
        }
        return a;
    }
    
    public static void main(String[] args) {
        int a = 48, b = 18;
        System.out.println("GCD of " + a + " and " + b + ": " + gcd(a, b));
        System.out.println("LCM of " + a + " and " + b + ": " + lcm(a, b));
    }
}

5. Power Calculation

public class PowerCalculation {
    // Iterative approach
    public static long power(int base, int exp) {
        long result = 1;
        for (int i = 0; i < exp; i++) {
            result *= base;
        }
        return result;
    }
    
    // Optimized using exponentiation by squaring
    public static long powerOptimized(int base, int exp) {
        if (exp == 0) return 1;
        
        long result = 1;
        long currentPower = base;
        
        while (exp > 0) {
            if (exp % 2 == 1) {
                result *= currentPower;
            }
            currentPower *= currentPower;
            exp /= 2;
        }
        
        return result;
    }
    
    public static void main(String[] args) {
        int base = 2, exp = 10;
        System.out.println(base + "^" + exp + " = " + powerOptimized(base, exp));
    }
}

Prime Number Problems

6. Check Prime Number

public class PrimeCheck {
    public static boolean isPrime(int num) {
        if (num <= 1) return false;
        if (num <= 3) return true;
        if (num % 2 == 0 || num % 3 == 0) return false;
        
        for (int i = 5; i * i <= num; i += 6) {
            if (num % i == 0 || num % (i + 2) == 0) {
                return false;
            }
        }
        
        return true;
    }
    
    public static void main(String[] args) {
        int[] numbers = {2, 17, 25, 29, 100};
        
        for (int num : numbers) {
            System.out.println(num + " is prime: " + isPrime(num));
        }
    }
}

7. Generate Prime Numbers (Sieve of Eratosthenes)

public class SieveOfEratosthenes {
    public static void sieveOfEratosthenes(int n) {
        boolean[] prime = new boolean[n + 1];
        Arrays.fill(prime, true);
        
        for (int p = 2; p * p <= n; p++) {
            if (prime[p]) {
                for (int i = p * p; i <= n; i += p) {
                    prime[i] = false;
                }
            }
        }
        
        System.out.print("Prime numbers up to " + n + ": ");
        for (int i = 2; i <= n; i++) {
            if (prime[i]) {
                System.out.print(i + " ");
            }
        }
        System.out.println();
    }
    
    public static void main(String[] args) {
        sieveOfEratosthenes(30);
    }
}

8. Prime Factors

public class PrimeFactors {
    public static void primeFactors(int n) {
        System.out.print("Prime factors of " + n + ": ");
        
        // Handle factor 2
        while (n % 2 == 0) {
            System.out.print(2 + " ");
            n /= 2;
        }
        
        // Handle odd factors
        for (int i = 3; i <= Math.sqrt(n); i += 2) {
            while (n % i == 0) {
                System.out.print(i + " ");
                n /= i;
            }
        }
        
        // If n is still greater than 2, it's a prime
        if (n > 2) {
            System.out.print(n);
        }
        
        System.out.println();
    }
    
    public static void main(String[] args) {
        primeFactors(315);
    }
}

9. Twin Primes

public class TwinPrimes {
    public static boolean isPrime(int num) {
        if (num <= 1) return false;
        if (num <= 3) return true;
        if (num % 2 == 0 || num % 3 == 0) return false;
        
        for (int i = 5; i * i <= num; i += 6) {
            if (num % i == 0 || num % (i + 2) == 0) {
                return false;
            }
        }
        return true;
    }
    
    public static void findTwinPrimes(int limit) {
        System.out.println("Twin primes up to " + limit + ":");
        
        for (int i = 2; i <= limit - 2; i++) {
            if (isPrime(i) && isPrime(i + 2)) {
                System.out.println("(" + i + ", " + (i + 2) + ")");
            }
        }
    }
    
    public static void main(String[] args) {
        findTwinPrimes(50);
    }
}

Palindrome Numbers

10. Check Palindrome Number

public class PalindromeNumber {
    public static boolean isPalindrome(int num) {
        if (num < 0) return false;
        
        int original = num;
        int reversed = 0;
        
        while (num > 0) {
            reversed = reversed * 10 + num % 10;
            num /= 10;
        }
        
        return original == reversed;
    }
    
    // Optimized approach (reverse only half)
    public static boolean isPalindromeOptimized(int num) {
        if (num < 0 || (num % 10 == 0 && num != 0)) return false;
        
        int reversedHalf = 0;
        while (num > reversedHalf) {
            reversedHalf = reversedHalf * 10 + num % 10;
            num /= 10;
        }
        
        return num == reversedHalf || num == reversedHalf / 10;
    }
    
    public static void main(String[] args) {
        int[] numbers = {121, 12321, 123, -121};
        
        for (int num : numbers) {
            System.out.println(num + " is palindrome: " + isPalindromeOptimized(num));
        }
    }
}

Fibonacci Series

12. Fibonacci Series

public class FibonacciSeries {
    // Iterative approach
    public static void fibonacciIterative(int n) {
        int a = 0, b = 1;
        
        System.out.print("Fibonacci series: " + a + " " + b + " ");
        
        for (int i = 2; i < n; i++) {
            int next = a + b;
            System.out.print(next + " ");
            a = b;
            b = next;
        }
        System.out.println();
    }
    
    // Recursive approach
    public static int fibonacciRecursive(int n) {
        if (n <= 1) return n;
        return fibonacciRecursive(n - 1) + fibonacciRecursive(n - 2);
    }
    
    // Optimized using memoization
    public static int fibonacciMemo(int n, int[] memo) {
        if (n <= 1) return n;
        if (memo[n] != 0) return memo[n];
        
        memo[n] = fibonacciMemo(n - 1, memo) + fibonacciMemo(n - 2, memo);
        return memo[n];
    }
    
    public static void main(String[] args) {
        int n = 10;
        fibonacciIterative(n);
        
        System.out.print("Using recursion: ");
        for (int i = 0; i < n; i++) {
            System.out.print(fibonacciRecursive(i) + " ");
        }
        System.out.println();
    }
}

13. Check Fibonacci Number

public class CheckFibonacci {
    public static boolean isPerfectSquare(int n) {
        int sqrt = (int) Math.sqrt(n);
        return sqrt * sqrt == n;
    }
    
    public static boolean isFibonacci(int n) {
        // A number is Fibonacci if one of (5*n*n + 4) or (5*n*n - 4) is perfect square
        return isPerfectSquare(5 * n * n + 4) || isPerfectSquare(5 * n * n - 4);
    }
    
    public static void main(String[] args) {
        int[] numbers = {8, 13, 15, 21, 34};
        
        for (int num : numbers) {
            System.out.println(num + " is Fibonacci: " + isFibonacci(num));
        }
    }
}

Armstrong Numbers

14. Check Armstrong Number

public class ArmstrongNumber {
    public static boolean isArmstrong(int num) {
        int original = num;
        int digits = String.valueOf(num).length();
        int sum = 0;
        
        while (num > 0) {
            int digit = num % 10;
            sum += Math.pow(digit, digits);
            num /= 10;
        }
        
        return sum == original;
    }
    
    public static void main(String[] args) {
        int[] numbers = {153, 371, 9474, 123};
        
        for (int num : numbers) {
            System.out.println(num + " is Armstrong: " + isArmstrong(num));
        }
    }
}

Mathematical Series

16. Factorial

public class Factorial {
    // Iterative approach
    public static long factorialIterative(int n) {
        long result = 1;
        for (int i = 2; i <= n; i++) {
            result *= i;
        }
        return result;
    }
    
    // Recursive approach
    public static long factorialRecursive(int n) {
        if (n <= 1) return 1;
        return n * factorialRecursive(n - 1);
    }
    
    // Using BigInteger for large numbers
    public static BigInteger factorialBig(int n) {
        BigInteger result = BigInteger.ONE;
        for (int i = 2; i <= n; i++) {
            result = result.multiply(BigInteger.valueOf(i));
        }
        return result;
    }
    
    public static void main(String[] args) {
        int n = 10;
        System.out.println("Factorial of " + n + ": " + factorialIterative(n));
        System.out.println("Large factorial (20!): " + factorialBig(20));
    }
}

17. Perfect Numbers

public class PerfectNumber {
    public static boolean isPerfect(int num) {
        if (num <= 1) return false;
        
        int sum = 1; // 1 is always a divisor
        
        for (int i = 2; i <= Math.sqrt(num); i++) {
            if (num % i == 0) {
                sum += i;
                if (i != num / i) { // Avoid counting square root twice
                    sum += num / i;
                }
            }
        }
        
        return sum == num;
    }
    
    public static void findPerfectNumbers(int limit) {
        System.out.println("Perfect numbers up to " + limit + ":");
        
        for (int i = 1; i <= limit; i++) {
            if (isPerfect(i)) {
                System.out.print(i + " ");
            }
        }
        System.out.println();
    }
    
    public static void main(String[] args) {
        findPerfectNumbers(10000);
    }
}

18. Happy Numbers

public class HappyNumber {
    public static boolean isHappy(int n) {
        Set<Integer> seen = new HashSet<>();
        
        while (n != 1 && !seen.contains(n)) {
            seen.add(n);
            n = getSumOfSquares(n);
        }
        
        return n == 1;
    }
    
    private static int getSumOfSquares(int n) {
        int sum = 0;
        while (n > 0) {
            int digit = n % 10;
            sum += digit * digit;
            n /= 10;
        }
        return sum;
    }
    
    public static void main(String[] args) {
        int[] numbers = {1, 7, 10, 13, 19, 23};
        
        for (int num : numbers) {
            System.out.println(num + " is happy: " + isHappy(num));
        }
    }
}

Number System Conversions

19. Binary Conversion

public class BinaryConversion {
    public static String decimalToBinary(int decimal) {
        if (decimal == 0) return "0";
        
        StringBuilder binary = new StringBuilder();
        
        while (decimal > 0) {
            binary.append(decimal % 2);
            decimal /= 2;
        }
        
        return binary.reverse().toString();
    }
    
    public static int binaryToDecimal(String binary) {
        int decimal = 0;
        int power = 0;
        
        for (int i = binary.length() - 1; i >= 0; i--) {
            if (binary.charAt(i) == '1') {
                decimal += Math.pow(2, power);
            }
            power++;
        }
        
        return decimal;
    }
    
    public static void main(String[] args) {
        int decimal = 25;
        String binary = decimalToBinary(decimal);
        
        System.out.println("Decimal " + decimal + " to binary: " + binary);
        System.out.println("Binary " + binary + " to decimal: " + binaryToDecimal(binary));
    }
}

20. Number Base Conversions

public class BaseConversion {
    public static String convertToBase(int number, int base) {
        if (number == 0) return "0";
        
        StringBuilder result = new StringBuilder();
        String digits = "0123456789ABCDEF";
        
        while (number > 0) {
            result.append(digits.charAt(number % base));
            number /= base;
        }
        
        return result.reverse().toString();
    }
    
    public static int convertFromBase(String number, int base) {
        int result = 0;
        int power = 0;
        
        for (int i = number.length() - 1; i >= 0; i--) {
            char digit = number.charAt(i);
            int value;
            
            if (Character.isDigit(digit)) {
                value = digit - '0';
            } else {
                value = digit - 'A' + 10;
            }
            
            result += value * Math.pow(base, power);
            power++;
        }
        
        return result;
    }
    
    public static void main(String[] args) {
        int number = 255;
        
        System.out.println("Decimal " + number + " to:");
        System.out.println("Binary: " + convertToBase(number, 2));
        System.out.println("Octal: " + convertToBase(number, 8));
        System.out.println("Hexadecimal: " + convertToBase(number, 16));
    }
}

Advanced Number Problems

21. Catalan Numbers

public class CatalanNumbers {
    public static long catalan(int n) {
        if (n <= 1) return 1;
        
        long[] dp = new long[n + 1];
        dp[0] = dp[1] = 1;
        
        for (int i = 2; i <= n; i++) {
            for (int j = 0; j < i; j++) {
                dp[i] += dp[j] * dp[i - 1 - j];
            }
        }
        
        return dp[n];
    }
    
    public static void main(String[] args) {
        System.out.println("First 10 Catalan numbers:");
        for (int i = 0; i < 10; i++) {
            System.out.print(catalan(i) + " ");
        }
    }
}
