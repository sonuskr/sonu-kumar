Stack - Complete Guide

What is a Stack?

A stack is a linear data structure that follows the LIFO (Last In, First Out) principle. The last element added is the first one to be removed.

Key Characteristics:

  • LIFO: Last In, First Out principle
  • Operations: Push (add), Pop (remove), Peek (view top)
  • Single End Access: Only top element is accessible
  • Limited Memory: Stack memory is limited (unlike heap)

Time Complexity:

  • Push: O(1) - Add to top
  • Pop: O(1) - Remove from top
  • Peek: O(1) - View top element
  • Search: O(n) - May need to traverse

Stack Implementation:


// Using Array
class Stack {
    int[] arr;
    int top = -1;
    
    Stack(int size) {
        arr = new int[size];
    }
    
    void push(int x) {
        if (top == arr.length - 1) {
            System.out.println("Stack Overflow");
        } else {
            arr[++top] = x;
        }
    }
    
    int pop() {
        if (top == -1) {
            System.out.println("Stack Underflow");
            return -1;
        }
        return arr[top--];
    }
    
    int peek() {
        if (top == -1) {
            return -1;
        }
        return arr[top];
    }
    
    boolean isEmpty() {
        return top == -1;
    }
}

// Java's Built-in Stack
Stack stack = new Stack<>();
stack.push(10);
stack.push(20);
stack.push(30);
int top = stack.pop();        // 30
int peek = stack.peek();       // 20
boolean empty = stack.isEmpty(); // false

Example: Valid Parentheses


public boolean isValid(String s) {
    Stack stack = new Stack<>();
    
    for (char c : s.toCharArray()) {
        if (c == '(' || c == '{' || c == '[') {
            stack.push(c);
        } else {
            if (stack.isEmpty()) return false;
            
            char top = stack.pop();
            if ((c == ')' && top != '(') ||
                (c == '}' && top != '{') ||
                (c == ']' && top != '[')) {
                return false;
            }
        }
    }
    return stack.isEmpty();
}

Example: Min Stack (with O(1) min retrieval)


class MinStack {
    Stack stack;
    Stack minStack;
    
    MinStack() {
        stack = new Stack<>();
        minStack = new Stack<>();
    }
    
    void push(int x) {
        stack.push(x);
        if (minStack.isEmpty() || x <= minStack.peek()) {
            minStack.push(x);
        }
    }
    
    void pop() {
        if (stack.pop().equals(minStack.peek())) {
            minStack.pop();
        }
    }
    
    int getMin() {
        return minStack.peek();
    }
}

Example: Next Greater Element


public int[] nextGreaterElement(int[] nums) {
    int[] result = new int[nums.length];
    Stack stack = new Stack<>();
    
    for (int i = nums.length - 1; i >= 0; i--) {
        // Pop all smaller elements
        while (!stack.isEmpty() && stack.peek() <= nums[i]) {
            stack.pop();
        }
        
        result[i] = stack.isEmpty() ? -1 : stack.peek();
        stack.push(nums[i]);
    }
    return result;
}

Disadvantages:
